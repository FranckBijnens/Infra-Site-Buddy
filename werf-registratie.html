📋 Google Sheets Installatiegids

Stap 1: Google Sheet aanmaken
Maak één nieuw, leeg Google Spreadsheet aan op sheets.google.com. De tabbladen Medewerkers, Werven, Bezoektypes en Registraties worden bij de eerste sync automatisch aangemaakt — je hoeft zelf niets in te vullen.

Stap 2: Apps Script openen
In het spreadsheet: Extensies → Apps Script. Vervang alle code door het script hieronder.

/**************************************************************** * Verbraeken Werf Registratie — CENTRALE BACKEND (Google Apps Script) * --------------------------------------------------------------- * Plak deze volledige code in: Extensies -> Apps Script * (vervang ALLE bestaande code), klik op Implementeren -> * Nieuwe implementatie -> type "Web-app", uitvoeren als "Ik", * toegang "Iedereen". Kopieer de /exec-URL naar de app * (Beheer -> Google Sheets Setup -> Apps Script URL). * * >>> BELANGRIJK: vul hieronder bij SPREADSHEET_ID het ID van je * Google Sheet in (de lange reeks in de URL tussen /d/ en /edit). * Zo weet het script altijd zeker met welke Sheet het moet werken. * * De tabbladen (Medewerkers, Werven, Bezoektypes, Registraties) worden * automatisch aangemaakt. Bij de eerste keer wordt een * standaard-admin aangemaakt: login: admin wachtwoord: admin123 * (wijzig dit nadien via Beheer of rechtstreeks in het tabblad). ****************************************************************/ // Plak hier het ID van je Google Sheet (tussen de aanhalingstekens). // Voorbeeld URL: https://docs.google.com/spreadsheets/d/1AbCdE... /edit -> ID = 1AbCdE... var SPREADSHEET_ID = ''; // Geeft de juiste spreadsheet terug: via ID indien ingevuld, anders de gekoppelde Sheet. function ss_() { return SPREADSHEET_ID ? var HEADERS = { users: ['id', 'name', 'login', 'pass', 'role'], sites: ['id', 'name', 'location', 'client'], types: ['id', 'icon', 'nl', 'fr', 'de', 'sub_nl', 'sub_fr', 'sub_de'], records: ['id', 'datum', 'tijd', 'vertrek_tijd', 'duur', 'gebruikerId', 'gebruiker', 'type', 'werf', 'bezoektype', 'bezoektypeLabel', 'gps_lat', 'gps_lon', 'gps_vertrek_lat', 'gps_vertrek_lon', 'opmerkingen'] }; // Standaard-bezoektypes (worden bij de eerste keer in het tabblad Bezoektypes gezet). var DEFAULT_TYPES = [ ['voorbereiding', '🗂️', 'Voorbereiding werf', 'Préparation chantier', 'Baustellenvorbereitung', 'Eerste bezoek & planning', 'Première visite & planification', 'Erster Besuch & Planung'], ['absteckung', '📍', 'Absteckung (GPS)', 'Absteckung (GPS)', 'Absteckung (GPS)', 'Uitmeten met GPS coördinaten', 'Mesure avec coordonnées GPS', 'GPS-Koordinaten abstecken'], ['platform-opbouw', '🏗️', 'Platform opbouw', 'Construction plateforme', 'Plattformaufbau', 'Opbouw van het werkplatform', 'Construction de la plateforme', 'Aufbau der Arbeitsplattform'], ['voortgang', '📊', 'Voortgang controle', 'Contrôle avancement', 'Fortschrittskontrolle', 'Algemene voortgang werf', 'Avancement général du chantier', 'Allgemeiner Baufortschritt'], ['veiligheid', '🦺', 'Veiligheid & PBM', 'Sécurité & EPI', 'Sicherheit & PSA', 'Controle veiligheid & PBM', 'Contrôle sécurité & EPI', 'Sicherheitskontrolle & PSA'], ['boorploeg', '⚙️', 'Boorploeg activiteiten', 'Équipe de forage', 'Bohrteam-Aktivitäten', 'Opvolging boorwerkzaamheden', 'Suivi des travaux de forage', 'Überwachung der Bohrarbeiten'], ['sanitair-burelen', '🚽', 'Sanitair & burelen', 'Sanitaires & bureaux', 'Sanitär & Büros', 'Controle voorzieningen', 'Vérification des installations', 'Kontrolle der Einrichtungen'], ['hekkens', '🚧', 'Werfhekkens', 'Clôtures de chantier', 'Bauzäune', 'Controle werfafscherming', 'Contrôle des clôtures', 'Kontrolle der Bauabsperrung'], ['terugbouw', '🔄', 'Terugbouw werf', 'Démontage chantier', 'Rückbau Baustelle', 'Opvolging afbraak en opruimen', 'Suivi du démontage', 'Abbau und Aufräumen'], ['overgave-tennet', '✅', 'Overgave aan Tennet', 'Remise à Tennet', 'Übergabe an Tennet', 'Finale inspectie en overdracht', 'Inspection finale et transfert', 'Endinspektion und Übergabe'] ]; function sheet_(name, headers) { var ss = ss_(); var sh = ss.getSheetByName(name) || ss.insertSheet(name); if (sh.getLastRow() === 0) { sh.appendRow(headers); sh.getRange(1, 1, 1, headers.length).setFontWeight('bold'); sh.setFrozenRows(1); } else { // Migratie: ontbrekende kolommen (zoals 'duur') achteraan toevoegen var existing = sh.getRange(1, 1, 1, sh.getLastColumn()).getValues()[0]; var missing = headers.filter(function (h) { return existing.indexOf(h) === -1; }); if (missing.length) { sh.getRange(1, existing.length + 1, 1, missing.length).setValues([missing]); sh.getRange(1, 1, 1, existing.length + missing.length).setFontWeight('bold'); } } return sh; } // Schrijf een rij weg op basis van de kolomnamen in de header (volgorde-onafhankelijk) function appendByHeader_(sh, obj) { var head = sh.getRange(1, 1, 1, sh.getLastColumn()).getValues()[0]; var row = head.map(function (h) { return (obj[h] !== undefined && obj[h] !== null) ? obj[h] : ''; }); sh.appendRow(row); } function readAll_(sh) { var values = sh.getDataRange().getValues(); if (values.length < 2) return []; var head = values[0]; var out = []; for (var i = 1; i < values.length; i++) { var o = {}; for (var c = 0; c < head.length; c++) o[head[c]] = values[i][c]; out.push(o); } return out; } function getUsers_() { var sh = sheet_(SHEETS.users, HEADERS.users); if (sh.getLastRow() === 1) { sh.appendRow([1, 'Admin Verbraeken', 'admin', 'admin123', 'admin']); } return readAll_(sh).map(function (u) { u.id = Number(u.id); return u; }); } function getSites_() { var sh = sheet_(SHEETS.sites, HEADERS.sites); return (v === '' || v === null || v === undefined) ? '' : String(v); } // Zet een celwaarde om naar 'HH:mm' (Google Sheets maakt van "08:00" een tijd op 30-12-1899) function timeStr_(v) { if (v instanceof Date) return Utilities.formatDate(v, Session.getScriptTimeZone(), 'HH:mm'); return (v === '' || v === null || v === undefined) ? '' : String(v); } function getRecords_() { var sh = sheet_(SHEETS.records, HEADERS.records); return readAll_(sh).map(function (r) { r.id = Number(r.id); r.gebruikerId = Number(r.gebruikerId); r.duur = Number(r.duur) || 0; r.datum = dateStr_(r.datum); r.tijd = timeStr_(r.tijd); r.vertrek_tijd = timeStr_(r.vertrek_tijd); r.gps_lat = (r.gps_lat === '' || r.gps_lat === null || r.gps_lat === undefined) ? '' : String(r.gps_lat); r.gps_lon = (r.gps_lon === '' || r.gps_lon === null || r.gps_lon === undefined) ? '' : String(r.gps_lon); r.gps_vertrek_lat = (r.gps_vertrek_lat === '' || r.gps_vertrek_lat === null || r.gps_vertrek_lat === undefined) ? '' : String(r.gps_vertrek_lat); r.gps_vertrek_lon = (r.gps_vertrek_lon === '' || r.gps_vertrek_lon === null || r.gps_vertrek_lon === undefined) ? '' : String(r.gps_vertrek_lon); return r; }); } function json_(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { try { var action = (e && e.parameter && e.parameter.action) || 'getAll'; if (action === 'getAll') { return json_({ ok: true, backend: BACKEND_VERSION, users: getUsers_(), sites: getSites_(), types: getTypes_(), records: getRecords_() }); } return json_({ ok: true, backend: BACKEND_VERSION }); } catch (err) { return json_({ ok: false, backend: BACKEND_VERSION, error: String(err) }); } } function doPost(e) { var lock = LockService.getScriptLock(); lock.waitLock(20000); try { var d = JSON.parse(e.postData.contents); var a = d.action; if (a === 'addRecord') { var r = d.record || {}; var rs = sheet_(SHEETS.records, HEADERS.records); if (!idExists_(rs, r.id)) appendByHeader_(rs, r); } else if (a === 'addUser') { var u = d.user || {}; getUsers_(); // zorgt dat sheet + standaard-admin bestaan var us = sheet_(SHEETS.users, HEADERS.users); if (!idExists_(us, u.id)) appendByHeader_(us, u); } else if (a === 'deleteUser') { deleteById_(SHEETS.users, d.id); } else if (a === 'addSite') { var s = d.site || {}; var ss2 = sheet_(SHEETS.sites, HEADERS.sites); if (!idExists_(ss2, s.id)) appendByHeader_(ss2, s); } else if (a === 'deleteSite') { deleteById_(SHEETS.sites, d.id); } else if (a === 'addType') { var tp = d.type || {}; getTypes_(); // zorgt dat sheet + standaardtypes bestaan var ts = sheet_(SHEETS.types, HEADERS.types); if (!idExists_(ts, tp.id)) appendByHeader_(ts, tp); } else if (a === 'deleteType') { deleteById_(SHEETS.types, d.id); } return json_({ ok: true }); } catch (err) { return json_({ ok: false, error: String(err) }); } finally { lock.releaseLock(); } } function deleteById_(name, id) { var ss = ss_(); var sh = ss.getSheetByName(name); if (!sh) return; var values = sh.getDataRange().getValues(); for (var i = values.length - 1; i >= 1; i--) { if (String(values[i][0]) === String(id)) sh.deleteRow(i + 1); } } function idExists_(sh, id) { if (sh.getLastRow() < 2) return false; var ids = sh.getRange(2, 1, sh.getLastRow() - 1, 1).getValues(); for (var i = 0; i < ids.length; i++) { if (String(ids[i][0]) === String(id)) return true; } return false; }

Stap 3: Implementeren als web-app
Klik op "Implementeren" → "Nieuwe implementatie". Type: Web-app. Uitvoeren als: Ik (jouw Google account). Toegang: Iedereen. Klik "Implementeren" en kopieer de URL.

Stap 4: Configureren in de app
Ga naar Beheer → Google Sheets Setup. Plak de Apps Script URL (de Spreadsheet ID is optioneel). Klik Opslaan — de app haalt meteen de centrale gegevens op.

✓ Vanaf nu delen alle toestellen dezelfde medewerkers, werven en registraties. Wijzigingen worden in twee richtingen gesynchroniseerd; zonder verbinding gemaakte registraties worden automatisch nagestuurd.

Standaard-admin bij eerste gebruik: admin / admin123 — wijzig dit nadien. Beheer medewerkers en werven vanaf één toestel; ze verschijnen dan op alle andere.
